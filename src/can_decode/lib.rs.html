1//! # can_decode
2//!
3//! Decode and encode CAN frames into messages/signals in a fast and easy way.
4//!
5//! ## Features
6//!
7//! - Parse DBC (CAN Database) files
8//! - Decode CAN messages into signals with physical values
9//! - Encode signal values back into raw CAN messages
10//! - Support for both standard and extended CAN IDs
11//! - Handle big-endian and little-endian byte ordering
12//! - Support for signed and unsigned signal values
13//! - Apply scaling factors and offsets (and inverse for encoding)
14//!
15//! ## Decoding Example
16//!
17//! ```no_run
18//! use can_decode::Parser;
19//! use std::path::Path;
20//!
21//! # fn main() -> Result<(), Box<dyn std::error::Error>> {
22//! // Load a DBC file
23//! let parser = Parser::from_dbc_file(Path::new("my_can_database.dbc"))?;
24//!
25//! // Decode a CAN message
26//! let msg_id = 0x123;
27//! let data = [0x12, 0x34, 0x56, 0x78, 0x9A, 0xBC, 0xDE, 0xF0];
28//!
29//! if let Some(decoded) = parser.decode_msg(msg_id, &data) {
30//!     println!("Message: {}", decoded.name);
31//!     for (signal_name, signal) in &decoded.signals {
32//!         println!("  {}: {} {}", signal_name, signal.value, signal.unit);
33//!     }
34//! }
35//! # Ok(())
36//! # }
37//! ```
38//!
39//! ## Encoding Example
40//!
41//! ```no_run
42//! use can_decode::Parser;
43//! use std::path::Path;
44//! use std::collections::HashMap;
45//!
46//! # fn main() -> Result<(), Box<dyn std::error::Error>> {
47//! let parser = Parser::from_dbc_file(Path::new("my_can_database.dbc"))?;
48//!
49//! // Encode a CAN message from signal values
50//! let mut signal_values = HashMap::from([
51//!     ("EngineSpeed".to_string(), 2500.0),
52//!     ("ThrottlePosition".to_string(), 45.5),
53//! ]);
54//!
55//! if let Some(data) = parser.encode_msg(0x123, &signal_values) {
56//!     println!("Encoded CAN data: {:02X?}", data);
57//! }
58//!
59//! // Or encode by message name
60//! if let Some((msg_id, data)) = parser.encode_msg_by_name("EngineData", &signal_values) {
61//!     println!("Message ID: {:#X}, Data: {:02X?}", msg_id, data);
62//! }
63//! # Ok(())
64//! # }
65//! ```

67/// A decoded CAN message containing signal values.
68///
69/// This structure represents a fully decoded CAN message with all its signals
70/// extracted and converted to physical values.
71#[derive(Debug, Clone)]
72pub struct DecodedMessage {
73    /// The name of the message as defined in the DBC file
74    pub name: String,
75    /// The CAN message ID
76    pub msg_id: u32,
77    /// Whether this is an extended (29-bit) CAN ID
78    pub is_extended: bool,
79    /// Transmitting node of the message ("Unknown" if not specified)
80    pub tx_node: String,
81    /// Map of signal names to their decoded values
82    pub signals: std::collections::HashMap<String, DecodedSignal>,
83}

99/// A CAN message parser that uses DBC file definitions.
100///
101/// The parser loads message and signal definitions from DBC files and uses them
102/// to decode raw CAN frame data into structured messages with physical signal values.
103///
104/// # Example
105///
106/// ```no_run
107/// use can_decode::Parser;
108/// use std::path::Path;
109///
110/// # fn main() -> Result<(), Box<dyn std::error::Error>> {
111/// let mut parser = Parser::new();
112/// parser.add_from_dbc_file(Path::new("engine.dbc"))?;
113/// parser.add_from_dbc_file(Path::new("transmission.dbc"))?;
114///
115/// let data = [0x00, 0x10, 0x20, 0x30, 0x40, 0x50, 0x60, 0x70];
116/// if let Some(decoded) = parser.decode_msg(0x100, &data) {
117///     println!("Decoded message: {}", decoded.name);
118/// }
119/// # Ok(())
120/// # }
121/// ```
122pub struct Parser {
123    /// Map of message ID to message definitions
124    msg_defs: std::collections::HashMap<u32, can_dbc::Message>,
125}

176    /// Adds message definitions from a DBC file string.
177    ///
178    /// This method parses DBC content from a string slice and adds all message
179    /// definitions to the parser. If a message ID already exists, it will be
180    /// overwritten and a warning will be logged.
181    ///
182    /// # Arguments
183    ///
184    /// * `buffer` - String slice containing the full DBC file contents
185    ///
186    /// # Errors
187    ///
188    /// Returns an error if the DBC content cannot be parsed.
189    ///
190    /// # Example
191    ///
192    /// ```no_run
193    /// use can_decode::Parser;
194    ///
195    /// # fn main() -> Result<(), Box<dyn std::error::Error>> {
196    /// let dbc_content = "VERSION \"\"..."; // DBC file content as &str
197    /// let mut parser = Parser::new();
198    /// parser.add_from_str(dbc_content)?;
199    /// # Ok(())
200    /// # }
201    /// ```
202    pub fn add_from_str(&mut self, buffer: &str) -> Result<(), Box<dyn std::error::Error>> {
203        let dbc = can_dbc::Dbc::try_from(buffer).map_err(|e| {
204            log::error!("Failed to parse DBC: {:?}", e);
205            format!("{:?}", e)
206        })?;
207        for msg_def in dbc.messages {
208            let msg_id = match msg_def.id {
209                can_dbc::MessageId::Standard(id) => id as u32,
210                can_dbc::MessageId::Extended(id) => id,
211            };
212            if self.msg_defs.contains_key(&msg_id) {
213                log::warn!(
214                    "Duplicate message ID {msg_id:#X} ({}). Overwriting existing definition.",
215                    msg_def.name
216                );
217            }
218            self.msg_defs.insert(msg_id, msg_def.clone());
219        }
220        Ok(())
221    }

260    /// Decodes a raw CAN message into structured data.
261    ///
262    /// Takes a CAN message ID and raw data bytes, then decodes all signals
263    /// according to the DBC definitions. Each signal is extracted, scaled,
264    /// and converted to its physical value.
265    ///
266    /// # Arguments
267    ///
268    /// * `msg_id` - The CAN message identifier
269    /// * `data` - The raw message data bytes (typically 0-8 bytes for standard CAN)
270    ///
271    /// # Returns
272    ///
273    /// Returns `Some(DecodedMessage)` if the message ID is known, or `None` if
274    /// the message ID is not found in the loaded DBC definitions.
275    ///
276    /// # Example
277    ///
278    /// ```no_run
279    /// use can_decode::Parser;
280    /// use std::path::Path;
281    ///
282    /// # fn main() -> Result<(), Box<dyn std::error::Error>> {
283    /// let parser = Parser::from_dbc_file(Path::new("my_database.dbc"))?;
284    ///
285    /// let msg_id = 0x123;
286    /// let data = [0x12, 0x34, 0x56, 0x78, 0x9A, 0xBC, 0xDE, 0xF0];
287    ///
288    /// if let Some(decoded) = parser.decode_msg(msg_id, &data) {
289    ///     println!("Message: {} (ID: {:#X})", decoded.name, decoded.msg_id);
290    ///     for (name, signal) in &decoded.signals {
291    ///         println!("  {}: {} {}", name, signal.value, signal.unit);
292    ///     }
293    /// } else {
294    ///     println!("Unknown message ID: {:#X}", msg_id);
295    /// }
296    /// # Ok(())
297    /// # }
298    /// ```
299    pub fn decode_msg(&self, msg_id: u32, data: &[u8]) -> Option<DecodedMessage> {
300        // Grab msg metadata and then for every signal in the message, decode it and add
301        // to the decoded message
302        let msg_def = self.msg_defs.get(&msg_id)?;
303        let is_extended = matches!(msg_def.id, can_dbc::MessageId::Extended(_));
304        let tx_node = match &msg_def.transmitter {
305            can_dbc::Transmitter::NodeName(name) => name.clone(),
306            can_dbc::Transmitter::VectorXXX => "Unknown".to_string(),
307        };
308        let mut decoded_signals = std::collections::HashMap::new();
309
310        for signal_def in &msg_def.signals {
311            match self.decode_signal(signal_def, data) {
312                Some(decoded_signal) => {
313                    decoded_signals.insert(decoded_signal.name.to_string(), decoded_signal);
314                }
315                None => {
316                    log::error!(
317                        "Failed to decode signal {} from message {}",
318                        signal_def.name,
319                        msg_def.name
320                    );
321                    return None;
322                }
323            }
324        }
325
326        Some(DecodedMessage {
327            name: msg_def.name.clone(),
328            msg_id,
329            is_extended,
330            tx_node,
331            signals: decoded_signals,
332        })
333    }

335    /// Decodes a single signal from raw CAN data.
336    ///
337    /// Extracts the raw bits for a signal, converts to signed/unsigned as needed,
338    /// and applies the scaling factor and offset to produce the physical value.
339    fn decode_signal(&self, signal_def: &can_dbc::Signal, data: &[u8]) -> Option<DecodedSignal> {
340        // Extract raw value based on byte order and signal properties
341        let raw_value = self.extract_signal_value(
342            data,
343            signal_def.start_bit as usize,
344            signal_def.size as usize,
345            signal_def.byte_order,
346        )?;
347
348        // Convert to signed if needed
349        let raw_value = if signal_def.value_type == can_dbc::ValueType::Signed {
350            // Convert to signed based on signal size
351            let max_unsigned = (1u64 << signal_def.size) - 1;
352            let sign_bit = 1u64 << (signal_def.size - 1);
353
354            if raw_value & sign_bit != 0 {
355                // Negative number - extend sign
356                (raw_value | (!max_unsigned)) as i64 as f64
357            } else {
358                raw_value as f64
359            }
360        } else {
361            raw_value as f64
362        };
363
364        // Apply scaling
365        let scaled_value = raw_value * signal_def.factor + signal_def.offset;
366
367        Some(DecodedSignal {
368            name: signal_def.name.clone(),
369            value: scaled_value,
370            unit: signal_def.unit.clone(),
371        })
372    }

374    /// Extracts raw signal bits from CAN data.
375    ///
376    /// Handles both little-endian and big-endian byte ordering according to
377    /// the signal definition.
378    fn extract_signal_value(
379        &self,
380        data: &[u8],
381        start_bit: usize,
382        size: usize,
383        byte_order: can_dbc::ByteOrder,
384    ) -> Option<u64> {
385        if data.is_empty() || size == 0 {
386            return None;
387        }
388
389        let total_bits = data.len() * 8;
390        if start_bit + size > total_bits {
391            return None;
392        }
393
394        let mut result = 0u64;
395
396        match byte_order {
397            can_dbc::ByteOrder::LittleEndian => {
398                let start_byte = start_bit / 8;
399                let start_bit_in_byte = start_bit % 8;
400
401                let mut remaining_bits = size;
402                let mut current_byte = start_byte;
403                let mut bit_offset = start_bit_in_byte;
404
405                while remaining_bits > 0 && current_byte < data.len() {
406                    let bits_in_this_byte = std::cmp::min(remaining_bits, 8 - bit_offset);
407                    let mask = ((1u64 << bits_in_this_byte) - 1) << bit_offset;
408                    let byte_value = ((data[current_byte] as u64) & mask) >> bit_offset;
409
410                    result |= byte_value << (size - remaining_bits);
411
412                    remaining_bits -= bits_in_this_byte;
413                    current_byte += 1;
414                    bit_offset = 0;
415                }
416            }
417            can_dbc::ByteOrder::BigEndian => {
418                // Big-endian (Motorola) bit extraction: iterate bits from
419                // start_bit toward higher bit positions, collecting each bit
420                // and appending into the result MSB-first.
421                let mut bit_pos = start_bit;
422
423                for _ in 0..size {
424                    let byte_idx = bit_pos / 8;
425                    let bit_idx = 7 - (bit_pos % 8);
426
427                    if byte_idx >= data.len() {
428                        break;
429                    }
430
431                    let bit_val = (data[byte_idx] >> bit_idx) & 1;
432                    result = (result << 1) | (bit_val as u64);
433
434                    bit_pos += 1;
435                }
436            }
437        }
438
439        Some(result)
440    }

442    /// Encodes a CAN message from signal values into raw bytes.
443    ///
444    /// Takes a message ID and a map of signal names to their physical values,
445    /// then encodes them according to the DBC definitions into raw CAN data bytes.
446    /// Applies inverse scaling (offset and factor) and packs bits according to
447    /// the signal's byte order and position. Applies scaling factors.
448    ///
449    /// # Arguments
450    ///
451    /// * `msg_id` - The CAN message identifier
452    /// * `signal_values` - Map of signal names to their physical values
453    ///
454    /// # Returns
455    ///
456    /// Returns `Some(Vec<u8>)` containing the encoded message data, or `None` if
457    /// the message ID is not found in the loaded DBC definitions or encoding fails.
458    ///
459    /// # Example
460    ///
461    /// ```no_run
462    /// use can_decode::Parser;
463    /// use std::path::Path;
464    /// use std::collections::HashMap;
465    ///
466    /// # fn main() -> Result<(), Box<dyn std::error::Error>> {
467    /// let parser = Parser::from_dbc_file(Path::new("my_database.dbc"))?;
468    ///
469    /// let signal_values = HashMap::from([
470    ///     ("EngineSpeed".to_string(), 2500.0),
471    ///     ("ThrottlePosition".to_string(), 45.5),
472    /// ]);
473    ///
474    /// if let Some(data) = parser.encode_msg(0x123, &signal_values) {
475    ///     println!("Encoded data: {:02X?}", data);
476    /// }
477    /// # Ok(())
478    /// # }
479    /// ```
480    pub fn encode_msg(
481        &self,
482        msg_id: u32,
483        signal_values: &std::collections::HashMap<String, f64>,
484    ) -> Option<Vec<u8>> {
485        let msg_def = self.msg_defs.get(&msg_id)?;
486
487        let msg_size = msg_def.size as usize;
488        let mut data = vec![0u8; msg_size];
489
490        for signal_def in &msg_def.signals {
491            let physical_value = match signal_values.get(&signal_def.name) {
492                Some(&v) => v,
493                None => {
494                    log::error!(
495                        "Signal {} not provided for message {} during encoding",
496                        signal_def.name,
497                        msg_def.name
498                    );
499                    return None;
500                }
501            };
502
503            // encode_signal() modifies the data buffer in place
504            if self
505                .encode_signal(signal_def, physical_value, &mut data)
506                .is_none()
507            {
508                log::error!(
509                    "Failed to encode signal {} for message {}",
510                    signal_def.name,
511                    msg_def.name
512                );
513                return None;
514            }
515        }
516
517        Some(data)
518    }

570    /// Encodes a single signal into raw CAN data.
571    ///
572    /// Applies inverse scaling (subtracts offset, divides by factor), converts to
573    /// the appropriate integer representation, and packs the bits into the data buffer.
574    fn encode_signal(
575        &self,
576        signal_def: &can_dbc::Signal,
577        physical_value: f64,
578        data: &mut [u8],
579    ) -> Option<()> {
580        // Apply inverse scaling: raw = (physical - offset) / factor
581        let raw_value = (physical_value - signal_def.offset) / signal_def.factor;
582
583        // Convert to integer and handle signed/unsigned
584        let raw_int = if signal_def.value_type == can_dbc::ValueType::Signed {
585            // Convert signed physical value to the integer representation
586            // then to the unsigned bit pattern (two's complement).
587            let signed_val = raw_value.round() as i64;
588            // For an N-bit signed value the allowed signed range is
589            // -(1 << (N-1)) .. (1 << (N-1)) - 1
590            let max_value = (1i64 << (signal_def.size - 1)) - 1;
591            let min_value = -(1i64 << (signal_def.size - 1));
592
593            // Clamp to valid signed range
594            let clamped = signed_val.max(min_value).min(max_value);
595
596            // Convert to unsigned representation (two's complement)
597            if clamped < 0 {
598                let mask = (1u64 << signal_def.size) - 1;
599                (clamped as u64) & mask
600            } else {
601                clamped as u64
602            }
603        } else {
604            // Unsigned value
605            let unsigned_val = raw_value.round() as u64;
606            let max_value = (1u64 << signal_def.size) - 1;
607
608            // Clamp to valid range
609            unsigned_val.min(max_value)
610        };
611
612        // Insert the value into the data buffer
613        self.insert_signal_value(
614            data,
615            signal_def.start_bit as usize,
616            signal_def.size as usize,
617            signal_def.byte_order,
618            raw_int,
619        )
620    }

622    /// Inserts raw signal bits into CAN data.
623    ///
624    /// Handles both little-endian and big-endian byte ordering according to
625    /// the signal definition.
626    fn insert_signal_value(
627        &self,
628        data: &mut [u8],
629        start_bit: usize,
630        size: usize,
631        byte_order: can_dbc::ByteOrder,
632        value: u64,
633    ) -> Option<()> {
634        if data.is_empty() || size == 0 {
635            return None;
636        }
637
638        let total_bits = data.len() * 8;
639        if start_bit + size > total_bits {
640            return None;
641        }
642
643        match byte_order {
644            can_dbc::ByteOrder::LittleEndian => {
645                let start_byte = start_bit / 8;
646                let start_bit_in_byte = start_bit % 8;
647
648                let mut remaining_bits = size;
649                let mut current_byte = start_byte;
650                let mut bit_offset = start_bit_in_byte;
651                let mut value_offset = 0;
652
653                while remaining_bits > 0 && current_byte < data.len() {
654                    let bits_in_this_byte = std::cmp::min(remaining_bits, 8 - bit_offset);
655                    let mask = ((1u8 << bits_in_this_byte) - 1) << bit_offset;
656
657                    // Extract bits from value
658                    let value_bits =
659                        ((value >> value_offset) & ((1u64 << bits_in_this_byte) - 1)) as u8;
660
661                    // Clear the bits in the data byte and set new bits
662                    data[current_byte] =
663                        (data[current_byte] & !mask) | ((value_bits << bit_offset) & mask);
664
665                    remaining_bits -= bits_in_this_byte;
666                    value_offset += bits_in_this_byte;
667                    current_byte += 1;
668                    bit_offset = 0;
669                }
670            }
671            can_dbc::ByteOrder::BigEndian => {
672                // Big-endian (Motorola) bit insertion: iterate bits from
673                // start_bit toward higher bit positions, extracting each bit
674                // from value MSB-first.
675                let mut bit_pos = start_bit;
676
677                for i in 0..size {
678                    let byte_idx = bit_pos / 8;
679                    let bit_idx = 7 - (bit_pos % 8);
680
681                    if byte_idx >= data.len() {
682                        break;
683                    }
684
685                    // Extract bit from value (MSB first)
686                    let bit_val = ((value >> (size - 1 - i)) & 1) as u8;
687
688                    // Clear the bit and set new value
689                    let mask = 1u8 << bit_idx;
690                    data[byte_idx] = (data[byte_idx] & !mask) | ((bit_val << bit_idx) & mask);
691
692                    bit_pos += 1;
693                }
694            }
695        }
696
697        Some(())
698    }

700    /// Returns all signal definitions for a given message ID.
701    ///
702    /// # Arguments
703    ///
704    /// * `msg_id` - The CAN message identifier
705    ///
706    /// # Returns
707    ///
708    /// Returns `Some(Vec<Signal>)` if the message ID is known, or `None` otherwise.
709    ///
710    /// # Example
711    ///
712    /// ```no_run
713    /// use can_decode::Parser;
714    /// use std::path::Path;
715    ///
716    /// # fn main() -> Result<(), Box<dyn std::error::Error>> {
717    /// let parser = Parser::from_dbc_file(Path::new("my_database.dbc"))?;
718    ///
719    /// if let Some(signals) = parser.signal_defs(0x123) {
720    ///     for signal in signals {
721    ///         println!("Signal: {}", signal.name);
722    ///     }
723    /// }
724    /// # Ok(())
725    /// # }
726    /// ```
727    pub fn signal_defs(&self, msg_id: u32) -> Option<Vec<can_dbc::Signal>> {
728        let msg_def = self.msg_defs.get(&msg_id)?;
729        Some(msg_def.signals.to_vec())
730    }

732    /// Returns all loaded message definitions.
733    ///
734    /// # Returns
735    ///
736    /// A vector containing all message definitions that have been loaded
737    /// from DBC files.
738    ///
739    /// # Example
740    ///
741    /// ```no_run
742    /// use can_decode::Parser;
743    /// use std::path::Path;
744    ///
745    /// # fn main() -> Result<(), Box<dyn std::error::Error>> {
746    /// let parser = Parser::from_dbc_file(Path::new("my_database.dbc"))?;
747    ///
748    /// for msg in parser.msg_defs() {
749    ///     println!("Message: {} (ID: {:#X})", msg.name,
750    ///              match msg.id {
751    ///                  can_dbc::MessageId::Standard(id) => id as u32,
752    ///                  can_dbc::MessageId::Extended(id) => id,
753    ///              });
754    /// }
755    /// # Ok(())
756    /// # }
757    /// ```
758    pub fn msg_defs(&self) -> Vec<can_dbc::Message> {
759        self.msg_defs.values().cloned().collect()
760    }
