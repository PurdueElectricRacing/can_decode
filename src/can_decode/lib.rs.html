1//! # can_decode
2//!
3//! Decode and encode CAN frames into messages/signals in a fast and easy way.
4//!
5//! ## Features
6//!
7//! - Parse DBC (CAN Database) files
8//! - Decode CAN messages into signals with physical values
9//! - Encode signal values back into raw CAN messages
10//! - Support for both standard and extended CAN IDs
11//! - Handle big-endian and little-endian byte ordering
12//! - Support for signed and unsigned signal values
13//! - Decode/encode IEEE-754 float signals (`SIG_VALTYPE_`)
14//! - Apply scaling factors and offsets (and inverse for encoding)
15//!
16//! ## Decoding Example
17//!
18//! ```no_run
19//! use can_decode::Parser;
20//! use std::path::Path;
21//!
22//! # fn main() -> Result<(), Box<dyn std::error::Error>> {
23//! // Load a DBC file
24//! let parser = Parser::from_dbc_file(Path::new("my_can_database.dbc"))?;
25//!
26//! // Decode a CAN message
27//! let msg_id = 0x123;
28//! let data = [0x12, 0x34, 0x56, 0x78, 0x9A, 0xBC, 0xDE, 0xF0];
29//!
30//! if let Some(decoded) = parser.decode_msg(msg_id, &data) {
31//!     println!("Message: {}", decoded.name);
32//!     for (signal_name, signal) in &decoded.signals {
33//!         println!("  {}: {:?} {}", signal_name, signal.value, signal.unit);
34//!     }
35//! }
36//! # Ok(())
37//! # }
38//! ```
39//!
40//! ## Encoding Example
41//!
42//! ```no_run
43//! use can_decode::Parser;
44//! use std::path::Path;
45//! use std::collections::HashMap;
46//!
47//! # fn main() -> Result<(), Box<dyn std::error::Error>> {
48//! let parser = Parser::from_dbc_file(Path::new("my_can_database.dbc"))?;
49//!
50//! // Encode a CAN message from signal values
51//! let signal_values = HashMap::from([
52//!     ("EngineSpeed".to_string(), 2500.0),
53//!     ("ThrottlePosition".to_string(), 45.5),
54//! ]);
55//!
56//! if let Some(data) = parser.encode_msg(0x123, &signal_values) {
57//!     println!("Encoded CAN data: {:02X?}", data);
58//! }
59//!
60//! // Or encode by message name
61//! if let Some((msg_id, data)) = parser.encode_msg_by_name("EngineData", &signal_values) {
62//!     println!("Message ID: {:#X}, Data: {:02X?}", msg_id, data);
63//! }
64//! # Ok(())
65//! # }
66//! ```

68/// Creates a bitmask with the lowest N bits set to 1.
69///
70/// This macro generates a mask value of the specified type with the lower `bits` bits set to 1
71/// and all other bits set to 0. It's used internally for extracting or masking the lower portion of
72/// integer values.
73///
74/// # Arguments
75///
76/// * `$bits` - The number of low bits to set (must be <= the bit width of the type)
77/// * `$ty` - The integer type for the mask (e.g., u8, u16, u32, u64)
78///
79/// # Examples
80///
81/// ```ignore
82/// let mask = low_bits_mask!(4, u8);  // Returns 0b00001111 (15)
83/// let mask = low_bits_mask!(8, u8);  // Returns 0b11111111 (255)
84/// let mask = low_bits_mask!(0, u16); // Returns 0
85/// ```
86///
87/// # Panics
88///
89/// Panics in debug builds if `bits` exceeds the bit width of the specified type.
90macro_rules! low_bits_mask {
91    ($bits:expr, $ty:ty) => {{
92        debug_assert!($bits <= <$ty>::BITS as usize);
93        match $bits {
94            0 => 0 as $ty,
95            n if n == <$ty>::BITS as usize => <$ty>::MAX,
96            n => <$ty>::MAX >> (<$ty>::BITS as usize - n),
97        }
98    }};
99}

107/// A decoded CAN message containing signal values.
108///
109/// This structure represents a fully decoded CAN message with all its signals
110/// extracted and converted to physical values.
111#[derive(Debug, Clone)]
112pub struct DecodedMessage {
113    /// The name of the message as defined in the DBC file
114    pub name: String,
115    /// The CAN message ID
116    pub msg_id: u32,
117    /// Whether this is an extended (29-bit) CAN ID
118    pub is_extended: bool,
119    /// Transmitting node of the message ("Unknown" if not specified)
120    pub tx_node: String,
121    /// Ordered map of signal names to their decoded values (maintains insertion order)
122    pub signals: SignalMap,
123}

136/// A decoded signal with its physical value.
137///
138/// Represents a single signal from a CAN message after decoding. The value is
139/// either a numeric physical value (after scaling/offset) or an enum label from

172/// A CAN message parser that uses DBC file definitions.
173///
174/// The parser loads message and signal definitions from DBC files and uses them
175/// to decode raw CAN frame data into structured messages with physical signal values.
176///
177/// # Example
178///
179/// ```no_run
180/// use can_decode::Parser;
181/// use std::path::Path;
182///
183/// # fn main() -> Result<(), Box<dyn std::error::Error>> {
184/// let mut parser = Parser::new();
185/// parser.add_from_dbc_file(Path::new("engine.dbc"))?;
186/// parser.add_from_dbc_file(Path::new("transmission.dbc"))?;
187///
188/// let data = [0x00, 0x10, 0x20, 0x30, 0x40, 0x50, 0x60, 0x70];
189/// if let Some(decoded) = parser.decode_msg(0x100, &data) {
190///     println!("Decoded message: {}", decoded.name);
191/// }
192/// # Ok(())
193/// # }
194/// ```
195pub struct Parser {
196    /// Map of message ID to message definitions
197    msg_defs: std::collections::HashMap<u32, can_dbc::Message>,
198
199    /// Map of message ID to per-signal enum/value-description mappings.
200    enum_defs: std::collections::HashMap<u32, Vec<EnumDef>>,
201
202    /// Map of message ID to per-signal IEEE float/double definitions.
203    float_defs: std::collections::HashMap<u32, Vec<FloatDef>>,
204}

257    /// Adds message and enum/value-description definitions from a DBC string.
258    ///
259    /// This method parses DBC content from a string slice and adds all message
260    /// definitions to the parser. If a message ID already exists, it will be
261    /// overwritten and a warning will be logged. Signal value descriptions
262    /// (enumerations) are also captured for enum decoding. Signal extended
263    /// value types (`SIG_VALTYPE_`) are captured for IEEE float/double decoding.
264    ///
265    /// # Arguments
266    ///
267    /// * `buffer` - String slice containing the full DBC file contents
268    ///
269    /// # Errors
270    ///
271    /// Returns an error if the DBC content cannot be parsed.
272    ///
273    /// # Example
274    ///
275    /// ```no_run
276    /// use can_decode::Parser;
277    ///
278    /// # fn main() -> Result<(), Box<dyn std::error::Error>> {
279    /// let dbc_content = "VERSION \"\"..."; // DBC file content as &str
280    /// let mut parser = Parser::new();
281    /// parser.add_from_str(dbc_content)?;
282    /// # Ok(())
283    /// # }
284    /// ```
285    pub fn add_from_str(&mut self, buffer: &str) -> Result<(), Box<dyn std::error::Error>> {
286        let dbc = can_dbc::Dbc::try_from(buffer).map_err(|e| {
287            log::error!("Failed to parse DBC: {:?}", e);
288            format!("{:?}", e)
289        })?;
290        for msg_def in dbc.messages {
291            let msg_id = msg_def.id.raw();
292            if self.msg_defs.contains_key(&msg_id) {
293                log::warn!(
294                    "Duplicate message ID {msg_id:#X} ({}). Overwriting existing definition.",
295                    msg_def.name
296                );
297            }
298            self.msg_defs.insert(msg_id, msg_def.clone());
299        }
300        for val_desc in dbc.value_descriptions {
301            match val_desc {
302                can_dbc::ValueDescription::Signal {
303                    message_id,
304                    name,
305                    value_descriptions,
306                } => {
307                    let msg_id = message_id.raw();
308                    let enum_def = EnumDef {
309                        signal_name: name.clone(),
310                        enum_map: value_descriptions
311                            .iter()
312                            .map(|vd| (vd.id, vd.description.clone()))
313                            .collect(),
314                    };
315                    let entry = self.enum_defs.entry(msg_id).or_default();
316                    if let Some(existing) = entry
317                        .iter_mut()
318                        .find(|e| e.signal_name == enum_def.signal_name)
319                    {
320                        *existing = enum_def;
321                        log::warn!(
322                            "Duplicate value description for signal '{}' in message ID {:#X}. \
323                            Overwriting existing enum definition.",
324                            name,
325                            msg_id
326                        );
327                    } else {
328                        entry.push(enum_def);
329                    }
330                }
331                can_dbc::ValueDescription::EnvironmentVariable { .. } => {}
332            }
333        }
334        for sig_ext_val_typ in dbc.signal_extended_value_type_list {
335            if sig_ext_val_typ.signal_extended_value_type
336                == can_dbc::SignalExtendedValueType::SignedOrUnsignedInteger
337            {
338                continue;
339            }
340
341            let msg_id = sig_ext_val_typ.message_id.raw();
342            let float_def = FloatDef {
343                signal_name: sig_ext_val_typ.signal_name.clone(),
344                float_def: sig_ext_val_typ.signal_extended_value_type,
345            };
346            let entry = self.float_defs.entry(msg_id).or_default();
347            if let Some(existing) = entry
348                .iter_mut()
349                .find(|e| e.signal_name == float_def.signal_name)
350            {
351                *existing = float_def.clone();
352                log::warn!(
353                    "Duplicate float definition for signal '{}' in message ID {:#X}. \
354                    Overwriting existing float definition.",
355                    float_def.signal_name,
356                    msg_id
357                );
358            } else {
359                entry.push(float_def);
360            }
361        }
362        Ok(())
363    }

402    /// Decodes a raw CAN message into structured data.
403    ///
404    /// Takes a CAN message ID and raw data bytes, then decodes all signals
405    /// according to the DBC definitions. Each signal is extracted, scaled,
406    /// and converted to its physical value or enum label (if defined).
407    ///
408    /// # Arguments
409    ///
410    /// * `msg_id` - The CAN message identifier
411    /// * `data` - The raw message data bytes (typically 0-8 bytes for standard CAN)
412    ///
413    /// # Returns
414    ///
415    /// Returns `Some(DecodedMessage)` if the message ID is known, or `None` if
416    /// the message ID is not found in the loaded DBC definitions.
417    ///
418    /// # Example
419    ///
420    /// ```no_run
421    /// use can_decode::Parser;
422    /// use std::path::Path;
423    ///
424    /// # fn main() -> Result<(), Box<dyn std::error::Error>> {
425    /// let parser = Parser::from_dbc_file(Path::new("my_database.dbc"))?;
426    ///
427    /// let msg_id = 0x123;
428    /// let data = [0x12, 0x34, 0x56, 0x78, 0x9A, 0xBC, 0xDE, 0xF0];
429    ///
430    /// if let Some(decoded) = parser.decode_msg(msg_id, &data) {
431    ///     println!("Message: {} (ID: {:#X})", decoded.name, decoded.msg_id);
432    ///     for (name, signal) in &decoded.signals {
433    ///         println!("  {}: {:?} {}", name, signal.value, signal.unit);
434    ///     }
435    /// } else {
436    ///     println!("Unknown message ID: {:#X}", msg_id);
437    /// }
438    /// # Ok(())
439    /// # }
440    /// ```
441    pub fn decode_msg(&self, msg_id: u32, data: &[u8]) -> Option<DecodedMessage> {
442        // Grab msg metadata and then for every signal in the message, decode it and add
443        // to the decoded message
444        let msg_def = self.msg_defs.get(&msg_id)?;
445        let is_extended = matches!(msg_def.id, can_dbc::MessageId::Extended(_));
446        let tx_node = match &msg_def.transmitter {
447            can_dbc::Transmitter::NodeName(name) => name.clone(),
448            can_dbc::Transmitter::VectorXXX => "Unknown".to_string(),
449        };
450        let mut decoded_signals = SignalMap::new();
451
452        for signal_def in &msg_def.signals {
453            match self.decode_signal(msg_id, signal_def, data) {
454                Some(decoded_signal) => {
455                    decoded_signals.insert(decoded_signal.name.to_string(), decoded_signal);
456                }
457                _ => {
458                    log::error!(
459                        "Failed to decode signal {} from message {}",
460                        signal_def.name,
461                        msg_def.name
462                    );
463                    return None;
464                }
465            }
466        }
467
468        Some(DecodedMessage {
469            name: msg_def.name.clone(),
470            msg_id,
471            is_extended,
472            tx_node,
473            signals: decoded_signals,
474        })
475    }

477    /// Decodes a single signal from raw CAN data.
478    ///
479    /// Extracts the raw bits for a signal, converts to signed/unsigned as needed,
480    /// and then either resolves a DBC enum label or applies scaling/offset to
481    /// produce a numeric physical value. If `SIG_VALTYPE_` marks the signal as
482    /// IEEE float/double, raw bits are interpreted directly as `f32`/`f64`.
483    fn decode_signal(
484        &self,
485        msg_id: u32,
486        signal_def: &can_dbc::Signal,
487        data: &[u8],
488    ) -> Option<DecodedSignal> {
489        // Extract raw value based on byte order and signal properties
490        let raw_value = self.extract_signal_value(
491            data,
492            signal_def.start_bit as usize,
493            signal_def.size as usize,
494            signal_def.byte_order,
495        )?;
496
497        // Convert to signed if needed
498        let raw_value_with_sign = if signal_def.value_type == can_dbc::ValueType::Signed {
499            // Convert to signed based on signal size
500            let max_unsigned = low_bits_mask!(signal_def.size as usize, u64);
501            let sign_bit = 1u64 << (signal_def.size - 1);
502
503            if raw_value & sign_bit != 0 {
504                // Negative number - extend sign
505                (raw_value | (!max_unsigned)) as i64
506            } else {
507                raw_value as i64
508            }
509        } else {
510            raw_value as i64
511        };
512
513        // Check if this signal has an enum definition
514        let enum_name = self
515            .enum_defs
516            .get(&msg_id)
517            .and_then(|enums| {
518                enums
519                    .iter()
520                    .find(|e| e.signal_name == signal_def.name)
521                    .and_then(|e| e.enum_map.get(&raw_value_with_sign))
522            })
523            .cloned();
524
525        if let Some(enum_str) = enum_name {
526            Some(DecodedSignal {
527                name: signal_def.name.clone(),
528                value: DecodedSignalValue::Enum(raw_value_with_sign, enum_str),
529                unit: signal_def.unit.clone(),
530            })
531        } else {
532            // Check for float definition
533            if let Some(float_def) = self.float_defs.get(&msg_id).and_then(|floats| {
534                floats
535                    .iter()
536                    .find(|f| f.signal_name == signal_def.name)
537                    .map(|f| &f.float_def)
538            }) {
539                // Interpret raw bits as float according to the definition
540                let float_value: Option<f64> = match float_def {
541                    can_dbc::SignalExtendedValueType::IEEEfloat32Bit => {
542                        if signal_def.size != 32 {
543                            log::warn!(
544                                "Signal {} marked as f32 but size is {} bits",
545                                signal_def.name,
546                                signal_def.size
547                            );
548                            return None;
549                        }
550
551                        Some(f32::from_bits(raw_value as u32) as f64)
552                    }
553
554                    can_dbc::SignalExtendedValueType::IEEEdouble64bit => {
555                        if signal_def.size != 64 {
556                            log::warn!(
557                                "Signal {} marked as f64 but size is {} bits",
558                                signal_def.name,
559                                signal_def.size
560                            );
561                            return None;
562                        }
563
564                        Some(f64::from_bits(raw_value))
565                    }
566
567                    _ => {
568                        unreachable!(
569                            "SignedOrUnsignedInteger should be filtered out when loading float defs"
570                        )
571                    }
572                };
573                if let Some(float_value) = float_value {
574                    let scaled_value = float_value * signal_def.factor + signal_def.offset;
575                    return Some(DecodedSignal {
576                        name: signal_def.name.clone(),
577                        value: DecodedSignalValue::Numeric(scaled_value),
578                        unit: signal_def.unit.clone(),
579                    });
580                }
581            }
582
583            // Apply scaling
584            let scaled_value = raw_value_with_sign as f64 * signal_def.factor + signal_def.offset;
585
586            Some(DecodedSignal {
587                name: signal_def.name.clone(),
588                value: DecodedSignalValue::Numeric(scaled_value),
589                unit: signal_def.unit.clone(),
590            })
591        }
592    }

594    /// Extracts raw signal bits from CAN data.
595    ///
596    /// Handles both little-endian and big-endian byte ordering according to
597    /// the signal definition.
598    fn extract_signal_value(
599        &self,
600        data: &[u8],
601        start_bit: usize,
602        size: usize,
603        byte_order: can_dbc::ByteOrder,
604    ) -> Option<u64> {
605        if data.is_empty() || size == 0 {
606            return None;
607        }
608
609        let mut result = 0u64;
610
611        match byte_order {
612            can_dbc::ByteOrder::LittleEndian => {
613                let start_byte = start_bit / 8;
614                let start_bit_in_byte = start_bit % 8;
615
616                let mut remaining_bits = size;
617                let mut current_byte = start_byte;
618                let mut bit_offset = start_bit_in_byte;
619
620                while remaining_bits > 0 {
621                    if current_byte >= data.len() {
622                        // Out of bounds: should not be considered a successful decode
623                        return None;
624                    }
625
626                    let bits_in_this_byte = std::cmp::min(remaining_bits, 8 - bit_offset);
627                    let mask = low_bits_mask!(bits_in_this_byte, u64) << bit_offset;
628                    let byte_value = ((data[current_byte] as u64) & mask) >> bit_offset;
629
630                    result |= byte_value << (size - remaining_bits);
631
632                    remaining_bits -= bits_in_this_byte;
633                    current_byte += 1;
634                    bit_offset = 0;
635                }
636            }
637            can_dbc::ByteOrder::BigEndian => {
638                // start_bit is the MSB position in DBC sawtooth numbering
639                // byte_idx = start_bit / 8, bit_in_byte = start_bit % 8 (counts from LSB of byte)
640                // Actual bit position within the byte = bit_in_byte (7=MSB, 0=LSB)
641
642                let start_byte = start_bit / 8;
643                let start_bit_in_byte = start_bit % 8; // Physical bit index
644
645                let mut byte_idx = start_byte;
646                let mut bit_in_byte = start_bit_in_byte as i32; // Counts down within byte
647
648                for _i in 0..size {
649                    if byte_idx >= data.len() {
650                        // Out of bounds: should not be considered a successful decode
651                        return None;
652                    }
653
654                    let bit_val = (data[byte_idx] >> bit_in_byte) & 1;
655                    result = (result << 1) | (bit_val as u64);
656
657                    bit_in_byte -= 1;
658                    if bit_in_byte < 0 {
659                        bit_in_byte = 7;
660                        byte_idx += 1;
661                    }
662                }
663            }
664        }
665
666        Some(result)
667    }

669    /// Encodes a CAN message from signal values into raw bytes.
670    ///
671    /// Takes a message ID and a map of signal names to their physical values,
672    /// then encodes them according to the DBC definitions into raw CAN data bytes.
673    /// Applies inverse scaling (offset and factor) and packs bits according to
674    /// the signal's byte order and position. Applies scaling factors.
675    ///
676    /// # Arguments
677    ///
678    /// * `msg_id` - The CAN message identifier
679    /// * `signal_values` - Map of signal names to their physical values
680    ///
681    /// # Returns
682    ///
683    /// Returns `Some(Vec<u8>)` containing the encoded message data, or `None` if
684    /// the message ID is not found in the loaded DBC definitions or encoding fails.
685    ///
686    /// # Example
687    ///
688    /// ```no_run
689    /// use can_decode::Parser;
690    /// use std::path::Path;
691    /// use std::collections::HashMap;
692    ///
693    /// # fn main() -> Result<(), Box<dyn std::error::Error>> {
694    /// let parser = Parser::from_dbc_file(Path::new("my_database.dbc"))?;
695    ///
696    /// let signal_values = HashMap::from([
697    ///     ("EngineSpeed".to_string(), 2500.0),
698    ///     ("ThrottlePosition".to_string(), 45.5),
699    /// ]);
700    ///
701    /// if let Some(data) = parser.encode_msg(0x123, &signal_values) {
702    ///     println!("Encoded data: {:02X?}", data);
703    /// }
704    /// # Ok(())
705    /// # }
706    /// ```
707    pub fn encode_msg(
708        &self,
709        msg_id: u32,
710        signal_values: &std::collections::HashMap<String, f64>,
711    ) -> Option<Vec<u8>> {
712        let msg_def = self.msg_defs.get(&msg_id)?;
713
714        let msg_size = msg_def.size as usize;
715        let mut data = vec![0u8; msg_size];
716
717        for signal_def in &msg_def.signals {
718            let physical_value = match signal_values.get(&signal_def.name) {
719                Some(&v) => v,
720                _ => {
721                    log::error!(
722                        "Signal {} not provided for message {} during encoding",
723                        signal_def.name,
724                        msg_def.name
725                    );
726                    return None;
727                }
728            };
729
730            // encode_signal() modifies the data buffer in place
731            if self
732                .encode_signal(msg_id, signal_def, physical_value, &mut data)
733                .is_none()
734            {
735                log::error!(
736                    "Failed to encode signal {} for message {}",
737                    signal_def.name,
738                    msg_def.name
739                );
740                return None;
741            }
742        }
743
744        Some(data)
745    }

747    /// Encodes a CAN message by message name instead of ID.
748    ///
749    /// Looks up the message by name and then encode it. This is slower as it
750    /// requires searching through all loaded messages. Applies scaling factors.
751    ///
752    /// # Arguments
753    ///
754    /// * `msg_name` - The name of the message as defined in the DBC file
755    /// * `signal_values` - Map of signal names to their physical values
756    ///
757    /// # Returns
758    ///
759    /// Returns `Some((msg_id, data))` containing the message ID and encoded data,
760    /// or `None` if the message name is not found or encoding fails.
761    ///
762    /// # Example
763    ///
764    /// ```no_run
765    /// use can_decode::Parser;
766    /// use std::path::Path;
767    /// use std::collections::HashMap;
768    ///
769    /// # fn main() -> Result<(), Box<dyn std::error::Error>> {
770    /// let parser = Parser::from_dbc_file(Path::new("my_database.dbc"))?;
771    ///
772    /// let signal_values = HashMap::from([
773    ///     ("EngineSpeed".to_string(), 2500.0),
774    ///     ("ThrottlePosition".to_string(), 45.5),
775    /// ]);
776    ///
777    /// if let Some((msg_id, data)) = parser.encode_msg_by_name("EngineData", &signal_values) {
778    ///     println!("Message ID: {:#X}, Data: {:02X?}", msg_id, data);
779    /// }
780    /// # Ok(())
781    /// # }
782    /// ```
783    pub fn encode_msg_by_name(
784        &self,
785        msg_name: &str,
786        signal_values: &std::collections::HashMap<String, f64>,
787    ) -> Option<(u32, Vec<u8>)> {
788        let (msg_id, _msg_def) = self
789            .msg_defs
790            .iter()
791            .find(|(_id, msg)| msg.name == msg_name)?;
792
793        let data = self.encode_msg(*msg_id, signal_values)?;
794        Some((*msg_id, data))
795    }

797    /// Encodes a single signal into raw CAN data.
798    ///
799    /// Applies inverse scaling (subtracts offset, divides by factor), converts to
800    /// the appropriate integer representation, and packs the bits into the data buffer.
801    fn encode_signal(
802        &self,
803        msg_id: u32,
804        signal_def: &can_dbc::Signal,
805        physical_value: f64,
806        data: &mut [u8],
807    ) -> Option<()> {
808        if let Some(float_def) = self.float_defs.get(&msg_id).and_then(|floats| {
809            floats
810                .iter()
811                .find(|f| f.signal_name == signal_def.name)
812                .map(|f| &f.float_def)
813        }) {
814            let raw_int = match float_def {
815                can_dbc::SignalExtendedValueType::IEEEfloat32Bit => {
816                    if signal_def.size != 32 {
817                        log::warn!(
818                            "Signal {} marked as f32 but size is {} bits",
819                            signal_def.name,
820                            signal_def.size
821                        );
822                        return None;
823                    }
824
825                    (physical_value as f32).to_bits() as u64
826                }
827
828                can_dbc::SignalExtendedValueType::IEEEdouble64bit => {
829                    if signal_def.size != 64 {
830                        log::warn!(
831                            "Signal {} marked as f64 but size is {} bits",
832                            signal_def.name,
833                            signal_def.size
834                        );
835                        return None;
836                    }
837
838                    physical_value.to_bits()
839                }
840
841                _ => {
842                    unreachable!(
843                        "SignedOrUnsignedInteger should be filtered out when loading float defs"
844                    )
845                }
846            };
847
848            return self.insert_signal_value(
849                data,
850                signal_def.start_bit as usize,
851                signal_def.size as usize,
852                signal_def.byte_order,
853                raw_int,
854            );
855        }
856
857        // Apply inverse scaling: raw = (physical - offset) / factor
858        let raw_value = (physical_value - signal_def.offset) / signal_def.factor;
859
860        // Convert to integer and handle signed/unsigned
861        let raw_int = if signal_def.value_type == can_dbc::ValueType::Signed {
862            // Convert signed physical value to the integer representation
863            // then to the unsigned bit pattern (two's complement).
864            let signed_val = raw_value.round() as i64;
865            // For an N-bit signed value the allowed signed range is
866            // -(1 << (N-1)) .. (1 << (N-1)) - 1
867            let max_value = (1i64 << (signal_def.size - 1)) - 1;
868            let min_value = -(1i64 << (signal_def.size - 1));
869
870            // Clamp to valid signed range
871            let clamped = signed_val.max(min_value).min(max_value);
872
873            // Convert to unsigned representation (two's complement)
874            if clamped < 0 {
875                let mask = low_bits_mask!(signal_def.size as usize, u64);
876                (clamped as u64) & mask
877            } else {
878                clamped as u64
879            }
880        } else {
881            // Unsigned value
882            let unsigned_val = raw_value.round() as u64;
883            let max_value = low_bits_mask!(signal_def.size as usize, u64);
884
885            // Clamp to valid range
886            unsigned_val.min(max_value)
887        };
888
889        // Insert the value into the data buffer
890        self.insert_signal_value(
891            data,
892            signal_def.start_bit as usize,
893            signal_def.size as usize,
894            signal_def.byte_order,
895            raw_int,
896        )
897    }

899    /// Inserts raw signal bits into CAN data.
900    ///
901    /// Handles both little-endian and big-endian byte ordering according to
902    /// the signal definition.
903    fn insert_signal_value(
904        &self,
905        data: &mut [u8],
906        start_bit: usize,
907        size: usize,
908        byte_order: can_dbc::ByteOrder,
909        value: u64,
910    ) -> Option<()> {
911        if data.is_empty() || size == 0 {
912            return None;
913        }
914
915        let total_bits = data.len() * 8;
916        if start_bit + size > total_bits {
917            return None;
918        }
919
920        match byte_order {
921            can_dbc::ByteOrder::LittleEndian => {
922                let start_byte = start_bit / 8;
923                let start_bit_in_byte = start_bit % 8;
924
925                let mut remaining_bits = size;
926                let mut current_byte = start_byte;
927                let mut bit_offset = start_bit_in_byte;
928                let mut value_offset = 0;
929
930                while remaining_bits > 0 && current_byte < data.len() {
931                    let bits_in_this_byte = std::cmp::min(remaining_bits, 8 - bit_offset);
932                    let mask = low_bits_mask!(bits_in_this_byte, u8) << bit_offset;
933
934                    // Extract bits from value
935                    let value_mask = low_bits_mask!(bits_in_this_byte, u64);
936                    let value_bits = ((value >> value_offset) & value_mask) as u8;
937
938                    // Clear the bits in the data byte and set new bits
939                    data[current_byte] =
940                        (data[current_byte] & !mask) | ((value_bits << bit_offset) & mask);
941
942                    remaining_bits -= bits_in_this_byte;
943                    value_offset += bits_in_this_byte;
944                    current_byte += 1;
945                    bit_offset = 0;
946                }
947            }
948            can_dbc::ByteOrder::BigEndian => {
949                let start_byte = start_bit / 8;
950                let start_bit_in_byte = start_bit % 8;
951
952                let mut byte_idx = start_byte;
953                let mut bit_in_byte = start_bit_in_byte as i32;
954
955                for i in 0..size {
956                    if byte_idx >= data.len() {
957                        return None;
958                    }
959
960                    let bit_val = ((value >> (size - 1 - i)) & 1) as u8;
961                    let mask = 1u8 << bit_in_byte;
962                    data[byte_idx] = (data[byte_idx] & !mask) | (bit_val << bit_in_byte);
963
964                    bit_in_byte -= 1;
965                    if bit_in_byte < 0 {
966                        bit_in_byte = 7;
967                        byte_idx += 1;
968                    }
969                }
970            }
971        }
972
973        Some(())
974    }
