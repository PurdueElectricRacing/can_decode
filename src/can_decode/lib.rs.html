35    pub fn add_from_slice(&mut self, buffer: &[u8]) -> Result<(), Box<dyn std::error::Error>> {
36        let dbc = can_dbc::DBC::from_slice(buffer).map_err(|e| {
37            log::error!("Failed to parse DBC: {:?}", e);
38            format!("{:?}", e)
39        })?;
40        for msg_def in dbc.messages() {
41            let msg_id = match msg_def.message_id() {
42                can_dbc::MessageId::Standard(id) => *id as u32,
43                can_dbc::MessageId::Extended(id) => *id,
44            };
45            if self.msg_defs.contains_key(&msg_id) {
46                log::warn!(
47                    "Duplicate message ID {msg_id:#X} ({}). Overwriting existing definition.",
48                    msg_def.message_name()
49                );
50            }
51            self.msg_defs.insert(msg_id, msg_def.clone());
52        }
53        Ok(())
54    }

68    pub fn decode_msg(&self, msg_id: IdType, data: &[u8]) -> Option<DecodedMessage> {
69        // Grab msg metadata and then for every signal in the message, decode it and add
70        // to the decoded message
71        let msg_def = self.msg_defs.get(&msg_id)?;
72        let msg_name = msg_def.message_name().to_string();
73        let is_extended = matches!(msg_def.message_id(), can_dbc::MessageId::Extended(_));
74        let mut decoded_signals = std::collections::HashMap::new();
75
76        for signal_def in msg_def.signals() {
77            match self.decode_signal(signal_def, data) {
78                Some(decoded_signal) => {
79                    decoded_signals.insert(decoded_signal.name.to_string(), decoded_signal);
80                }
81                None => {
82                    log::warn!(
83                        "Failed to decode signal {} from message {}",
84                        signal_def.name(),
85                        msg_name
86                    );
87                }
88            }
89        }
90
91        Some(DecodedMessage {
92            name: msg_name,
93            msg_id,
94            is_extended,
95            signals: decoded_signals,
96        })
97    }

99    fn decode_signal(&self, signal_def: &can_dbc::Signal, data: &[u8]) -> Option<DecodedSignal> {
100        // Get signal properties
101        let start_bit = *signal_def.start_bit() as usize;
102        let signal_size = *signal_def.signal_size() as usize;
103        let byte_order = signal_def.byte_order();
104        let value_type = signal_def.value_type();
105        let factor = signal_def.factor();
106        let offset = signal_def.offset();
107        let unit = signal_def.unit();
108
109        // Extract raw value based on byte order and signal properties
110        let raw_value = self.extract_signal_value(data, start_bit, signal_size, *byte_order)?;
111
112        // Convert to signed if needed
113        let raw_value = if *value_type == can_dbc::ValueType::Signed {
114            // Convert to signed based on signal size
115            let max_unsigned = (1u64 << signal_size) - 1;
116            let sign_bit = 1u64 << (signal_size - 1);
117
118            if raw_value & sign_bit != 0 {
119                // Negative number - extend sign
120                (raw_value | (!max_unsigned)) as i64 as f64
121            } else {
122                raw_value as f64
123            }
124        } else {
125            raw_value as f64
126        };
127
128        // Apply scaling
129        let scaled_value = raw_value * factor + offset;
130
131        Some(DecodedSignal {
132            name: signal_def.name().to_string(),
133            value: scaled_value,
134            unit: unit.to_string(),
135        })
136    }

138    fn extract_signal_value(
139        &self,
140        data: &[u8],
141        start_bit: usize,
142        size: usize,
143        byte_order: can_dbc::ByteOrder,
144    ) -> Option<u64> {
145        if data.is_empty() || size == 0 {
146            return None;
147        }
148
149        let total_bits = data.len() * 8;
150        if start_bit + size > total_bits {
151            return None;
152        }
153
154        let mut result = 0u64;
155
156        match byte_order {
157            can_dbc::ByteOrder::LittleEndian => {
158                let start_byte = start_bit / 8;
159                let start_bit_in_byte = start_bit % 8;
160
161                let mut remaining_bits = size;
162                let mut current_byte = start_byte;
163                let mut bit_offset = start_bit_in_byte;
164
165                while remaining_bits > 0 && current_byte < data.len() {
166                    let bits_in_this_byte = std::cmp::min(remaining_bits, 8 - bit_offset);
167                    let mask = ((1u64 << bits_in_this_byte) - 1) << bit_offset;
168                    let byte_value = ((data[current_byte] as u64) & mask) >> bit_offset;
169
170                    result |= byte_value << (size - remaining_bits);
171
172                    remaining_bits -= bits_in_this_byte;
173                    current_byte += 1;
174                    bit_offset = 0;
175                }
176            }
177            can_dbc::ByteOrder::BigEndian => {
178                // Idk if this is right
179                let mut bit_pos = start_bit;
180
181                for _ in 0..size {
182                    let byte_idx = bit_pos / 8;
183                    let bit_idx = 7 - (bit_pos % 8);
184
185                    if byte_idx >= data.len() {
186                        break;
187                    }
188
189                    let bit_val = (data[byte_idx] >> bit_idx) & 1;
190                    result = (result << 1) | (bit_val as u64);
191
192                    bit_pos += 1;
193                }
194            }
195        }
196
197        Some(result)
198    }
